const express = require('express'); const app = express(); const PORT = process.env.PORT || 3000; const fs = require('fs'); const files = fs.readdirSync('/app'); console.log('Files in /app:', files); app.get('*', (req, res) => { res.sendFile('/app/index.html'); }); app.listen(PORT, () => { console.log('Running on port ' + PORT); });
